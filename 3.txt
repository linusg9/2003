import nltk,numpy as np,matplotlib.pyplot as plt,seaborn as sns
from nltk.corpus import reuters
from gensim.models import Word2Vec
from sklearn.manifold import TSNE

nltk.download('punkt')
nltk.download('reuters')

sents=[[w.lower() for w in s if w.isalpha()] for s in reuters.sents()[:5000]]

m=Word2Vec(sents,vector_size=100,window=5,min_count=2)

print(m.wv.most_similar("disease",topn=5))

words=["health","disease","doctor","patient","treatment",
       "medicine","virus","surgery","nurse","hospital"]

v=np.array([m.wv[w] for w in words if w in m.wv])

p=TSNE(n_components=2,perplexity=3,random_state=42).fit_transform(v)

plt.figure(figsize=(10,6))
sns.scatterplot(x=p[:,0],y=p[:,1])

for i,w in enumerate([w for w in words if w in m.wv]):
    plt.annotate(w,(p[i,0],p[i,1]),
       bbox=dict(facecolor='white',alpha=0.7))

plt.title("t-SNE Visualization of Medical Word Embeddings")
plt.xlabel("t-SNE Dimension 1")
plt.ylabel("t-SNE Dimension 2")
plt.grid(True)
plt.show()
